package com.atcs.demo;

import java.util.*;

public class CounterLettersInArray {
//Main method
  public static void main(String args[]) {
    // Declare and create an array
     char[] arr = createArray();

    
    System.out.println("The lowercase letters are:");
     displayArray(arr);
    // Count the occurrences of each letter
    int[] counts = countLetters(arr);
    
    
    System.out.println();
    System.out.println("The occurrences of each letter are:");
     displayCounts(counts);
  }

//Create an array of characters 
  public static char[] createArray() {
    // Declare an array of characters and create it
    char[] arr = new char[100];
    Random generator = new Random();
    String S ="abcdefghijklmnopqrstuvwxyz";
    for (int i = 0; i < arr.length; i++) 
      arr[i] = (char) S.charAt(generator.nextInt(26));
   return arr;
  } 
 

//Display the array of characters 
  public static void displayArray(char[] arr) {
    // Display the characters in the array 20 on each line
    for (int i = 0; i < arr.length; i++) {
      if ((i + 1) % 20 == 0)
        System.out.println(arr[i] + " ");
      else
       System.out.print(arr[i] + " ");
           }
         }
      
//Count the occurrences of each letter 
         public static int[] countLetters(char[] arr) {
           // Declare and create an array of 26 int
           int[] counts = new int[26];
      
           // For each lowercase letter in the array, count it
           for (int i = 0; i < arr.length; i++) 
             counts[arr[i] - 'a']++;
      
           return counts;
         }
         
      
// Display counts
         public static void displayCounts(int[] counts) {
           for (int i = 0; i < counts.length; i++) {
               System.out.print(counts[i] + " " + (char)(i + 'a') + " ");
           }
         }
      }
